49  	/**
50  	 * get a String from a given URL
51  	 * @param urlString
52  	 * @return
53  	 */
54  	public String getStringFromUrl(String urlString) { 
55  		client = ApacheHttpClient.create();
56  		WebResource webResource = client.resource(urlString);
57  		ClientResponse response = webResource.get(ClientResponse.class);
58  
59  		if (response.getStatus() != 200) {
60  			throw new RuntimeException("HTTP error code : " + response.getStatus());
61  		}
62  		String result=response.getEntity(String.class);
63  		return result;
64  	}
65  	
66  	/**
67  	 * get the a CSV File from the given urlString
68  	 * http://stackoverflow.com/questions/27224870/getting-http-400-error-when-trying-to-download-file-using-jersey-client
69  	 * 
70  	 * @param urlString
71  	 * @param csvFile
72  	 * @throws IOException
73  	 */
74  	public void getCSVAsFile(String urlString, File csvFile) throws IOException {
75  		client = ApacheHttpClient.create();
76  		WebResource webResource = client.resource(urlString);
77  		WebResource.Builder wb = webResource
78  				.accept("application/json,application/pdf,text/plain,image/jpeg,application/xml,application/vnd.ms-excel");
79  		ClientResponse response = wb.get(ClientResponse.class);
80  
81  		if (response.getStatus() != 200) {
82  			throw new RuntimeException("HTTP error code : " + response.getStatus());
83  		}
84  
85  		InputStream input = response.getEntity(InputStream.class);
86  
87  		byte[] byteArray = org.apache.commons.io.IOUtils.toByteArray(input);
88  
89  		FileOutputStream fos = new FileOutputStream(csvFile);
90  		fos.write(byteArray);
91  		fos.flush();
92  		fos.close();
93  	}
