53     * @throws Exception
54     */
55    @Test
56    public void testGetRecentChanges() throws Exception {
57      Api api = getQueryResult(
58          "&list=recentchanges&rcprop=title%7Cids%7Csizes%7Cflags%7Cuser&rclimit=3");
59      List<Rc> rcList = api.getQuery().getRecentchanges();
60      assertEquals(3, rcList.size());
61      for (Rc rc : rcList) {
62        assertNotNull(rc.getTitle());
63      }
64    }

66    /**
67     * show a list of recent changes
68     * 
69     * @param rcList
70     */
71    public void showRCList(List<Rc> rcList) {
72      LOGGER.log(Level.INFO, "found " + rcList.size() + " recent changes");
73      for (Rc rc : rcList) {
74        LOGGER.log(Level.INFO,
75            rc.getTitle() + "(" + rc.getPageid() + "/" + rc.getTimestamp() + ")");
76      }
77    }

79    /**
80     * get recent changes specifying the number of days
81     * 
82     * @throws Exception
83     */
84    @Test
85    public void testGetRecentChangesWithDays() throws Exception {
86      // debug=true;
87      Mediawiki lwiki = super.getWiki().getMediaWikiJapi();
88      if (debug)
89        lwiki.setDebug(debug);
90      Date today = new Date();
91      Calendar cal = new GregorianCalendar();
92      cal.setTime(today);
93      cal.add(Calendar.DAY_OF_MONTH, -30);
94      Date date30daysbefore = cal.getTime();
95      String rcstart = lwiki.dateToMWTimeStamp(today);
96      String rcend = lwiki.dateToMWTimeStamp(date30daysbefore);
97      int rclimit = 4;
98      List<Rc> rcList = lwiki.getRecentChanges(rcstart, rcend, rclimit);
99      assertNotNull(rcList);
100 
101     if (debug) {
102       showRCList(rcList);
103 
104     }
105     assertTrue(rcList.size() > 0);
106   }

135   @Test
136   public void sortByTitleAndFilterDoubles() throws Exception {
137     List<Rc> rcList = new ArrayList<Rc>();
138     String[] titles = { "Title1", "Title2", "Title2", "Title1", "Title3" };
139     String[] ts = { "2015-01-01", "2015-01-01", "2016-01-02", "2016-01-01",
140         "2015-07-01" };
141     SimpleDateFormat tsFormat = new SimpleDateFormat("yyyy-MM-dd");
142     int index = 0;
143     for (String title : titles) {
144       Rc testRc = new Rc();
145       testRc.setTitle(title);
146       Date tsDate = tsFormat.parse(ts[index++]);
147       GregorianCalendar cal = new GregorianCalendar();
148       cal.setTime(tsDate);
149       XMLGregorianCalendar xdate = DatatypeFactory.newInstance()
150           .newXMLGregorianCalendar(cal);
151       testRc.setTimestamp(xdate);
152       rcList.add(testRc);
153     }
154     ExampleWiki lewiki = ewm.get("mediawiki-japi-test1_27");
155     Mediawiki lwiki = lewiki.getMediaWikiJapi();
156     // debug=true;
157     List<Rc> sortedFilteredList = lwiki.sortByTitleAndFilterDoubles(rcList);
158     if (debug) {
159       showRCList(rcList);
160       showRCList(sortedFilteredList);
161     }
