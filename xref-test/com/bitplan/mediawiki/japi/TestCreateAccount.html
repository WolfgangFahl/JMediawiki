40  public class TestCreateAccount extends APITestbase {
41  
42    /**
43     * test getting the create Account token
44     * 
45     * @throws Exception
46     */
47    @Test
48    public void testCreateAccount() throws Exception {
49      ExampleWiki lwiki = ewm.get("mediawiki-japi-test1_27");
50      Mediawiki wiki = (Mediawiki) lwiki.wiki;
51      if (hasWikiUser(lwiki)) {
52        lwiki.login();
53        // lwiki.wiki.setDebug(true);
54        DateFormat df = new SimpleDateFormat("yyyyMMddHHmmSS");
55        String nowAsISO = df.format(new Date());
56        Api api = wiki.createAccount("JohnDoe" + nowAsISO, "wf@bitplan.com",
57            "John%20Doe", true, "SMWCon2015-05", "en");
58        assertNull(api.getWarnings());
59        String result = api.getCreateaccount().getResult();
60        assertEquals("Success", result);
61      }
62    }
