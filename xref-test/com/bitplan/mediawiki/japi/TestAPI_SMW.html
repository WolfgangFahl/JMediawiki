35  /**
36   * test the API for Semantic Mediawiki 
37   */
38  public class TestAPI_SMW extends APITestbase {
39    
40    @Test
41    public void testBrowseBySubject() throws Exception {
42      // semantic-mediawiki.org has a certificate that
43      // is not in the chain understood by java
44      SSLWiki.ignoreCertificates=true;
45      SSLWiki wiki = new SSLWiki("https://www.semantic-mediawiki.org","/w");
46      String params="&subject=Demo:Amsterdam";
47      Api api=wiki.getActionResult("browsebysubject", params,null,null,"json");
48      assertNotNull("the API result should not be null",api);
49      Query query = api.getQuery();
50      assertNotNull(query);
51      assertNotNull(query.getSubject());
52      assertNotNull(query.getSerializer());
53      assertNotNull(query.getVersion());
54      assertNotNull(query.getData());
55      assertEquals("Amsterdam#202#",query.getSubject());
56      assertEquals("SMW\\Serializers\\SemanticDataSerializer",query.getSerializer());
57      assertEquals("0.1",query.getVersion());
58      List<Property> data = query.getData();
59      assertEquals(18,data.size());
60      for (Property prop:data) {
61        assertNotNull(prop.getDataitem());
62        for (DataItem item:prop.getDataitem()) {
63          assertNotNull(item.getType());
64          assertNotNull(item.getItem());
65          // debug=true;
66          if (debug)
67            System.out.println(prop.getProperty()+" "+item.getType()+"="+item.getItem());
68        }
69      }
70    }
71  }
