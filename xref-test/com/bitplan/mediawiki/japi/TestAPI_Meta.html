50  	@Test
51  	public void testGeneralSiteInfo() throws Exception {
52  		for (ExampleWiki lwiki : getWikis()) {
53  			// lwiki.wiki.setDebug(true);
54  		  // FIXME add to interface
55  			General general = lwiki.getGeneral();
56  			assertNotNull(general);
57  			boolean mayBeNull = true;
58  			check("generator", general.getGenerator());
59  			check("logo", general.getLogo(), mayBeNull);
60  			String wikiVersion=general.getGenerator();
61  			// debug=true;
62  			if (debug)
63  			  LOGGER.log(Level.INFO,wikiVersion);
64  			if (wikiVersion.compareToIgnoreCase("Mediawiki 1.20") >= 0) {
65  				assertEquals(lwiki.wiki.getSiteurl()+"/"+lwiki.wiki.getScriptPath(),lwiki.getLogo(), general.getLogo());
66  				check("favicon", general.getFavicon());
67  				check("langconversion", general.getLangconversion());
68  				check("linkprefix", general.getLinkprefix());
69  				check("linkprefixcharset", general.getLinkprefixcharset());
70  				check("linktrail", general.getLinktrail());
71  				// MediaWiki 1.27.0-wmf.20 does not have this
72  				// check("maxuploadsize", general.getMaxuploadsize());
73  				List<Limit> imageLimits = general.getImagelimits().getLimit();
74  				assertNotNull(imageLimits);
75  				for (Limit imageLimit : imageLimits) {
76  					check("\twidth", imageLimit.getWidth());
77  					check("\theight", imageLimit.getHeight());
78  				}
79  				Extensiondistributor extdist = general.getExtensiondistributor();
80  				if (extdist != null) {
81  					check("ext-list", extdist.getList());
82  					Snapshots snapshots = extdist.getSnapshots();
83  					assertNotNull(snapshots);
84  					for (String snapshot : snapshots.getSnapshot()) {
85  						check("snapshot", snapshot.trim());
86  					}
87  				}
88  				String generator=general.getGenerator();
89  				if (generator.compareToIgnoreCase("Mediawiki 1.25") >= 0) {
90  				  String githash=general.getGitHash();
91  					check("githash", githash,true);				
92  				}
93  			}
94  			check("language", general.getLang());
95  			check("articlepath", general.getArticlepath());
96  			check("base", general.getBase());
97  			check("case", general.getCase());
98  			check("dbtype", general.getDbtype());
99  			check("dbversion", general.getDbversion());
100 			check("fallback", general.getFallback());
101 			check("gitbranch", general.getGitBranch(), mayBeNull);
102 			check("hhvmversion", general.getHhvmversion(), mayBeNull);
103 			check("imagewhitelistenabled", general.getImagewhitelistenabled(),
104 					mayBeNull);
105 			check("legaltitlechars", general.getLegaltitlechars(), mayBeNull);
106 			check("mainpage", general.getMainpage());
107 			check("misermode", general.getMisermode(), mayBeNull);
108 			check("phpsapi", general.getPhpsapi());
109 			check("phpversion", general.getPhpversion());
110 			check("script", general.getScript());
111 			check("scriptpath", general.getScriptpath());
112 			check("server", general.getServer());
113 			check("servername", general.getServername(), mayBeNull);
114 			check("sitename", general.getSitename());
115 		}

119 	/**
120 	 * test statistics
121 	 * @throws Exception
122 	 */
123 	public void testStatistics() throws Exception {
124 		for (ExampleWiki lwiki : getWikis()) {
125 			Api api = getQueryResult(lwiki, "&meta=siteinfo&siprop=statistics");
126 			assertNotNull(api);
127 			Query query = api.getQuery();
128 			assertNotNull(query);
129 			Statistics statistics = query.getStatistics();
130 			assertNotNull(statistics);
131       assertTrue(
132           lwiki.wiki.getSiteurl() + lwiki.wiki.getScriptPath()+" " + statistics.getPages() + ">"
133               + lwiki.getExpectedPages(),
134           statistics.getPages().intValue() >= lwiki.getExpectedPages());
135 		}

139 	 * test namespace retrieving
140    * http://www.mediawiki.org/wiki/Help:Namespaces
141    * @throws Exception
142    */
143   @Test
144   public void testNameSpaces() throws Exception {
145     for (ExampleWiki lwiki : getWikis()) {
146       Mediawiki mw=(Mediawiki) lwiki.wiki;
147       Map<String, Ns> namespaces = mw.getSiteInfo().getNamespaces();
148       assertNotNull(namespaces);
149       assertTrue(namespaces.size()>12);
150       // debug=true;
151       if (debug) {
152         for (Ns namespace:namespaces.values()) {
153           LOGGER.log(Level.INFO,namespace.getId()+":"+namespace.getValue()+"->"+namespace.getCanonical());
154         }
155       }
156     }
157   }

190   @Test
191   public void testGetNameSpaceFromPageTitle() throws Exception {
192     String pageTitles[]={"Template:someTemplate","MainPage","Template:someTemplate:with weird colons:","File:SomeFilewith File: again"};
