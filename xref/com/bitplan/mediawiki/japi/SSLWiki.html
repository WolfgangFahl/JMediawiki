41  /**
42   * an SSL wiki
43   * 
44   * @author wf http://stackoverflow.com/questions/2703161/how-to-ignore-ssl-
45   *         certificate -errors-in-apache-httpclient-4-0
46   */
47  public class SSLWiki extends Mediawiki {
48    public static boolean ignoreCertificates = false;
49  
50    /**
51     * avoid ssl issue with hostname check
52     * 
53     * @author wf
54     *
55     */
56    public static class IgnoreHostName implements HostnameVerifier {
57      boolean debug = false;
58      protected static Logger LOGGER = Logger.getLogger("org.sidif.wiki");
59  
60      @Override
61      public boolean verify(String hostname, SSLSession sslSession) {
62        if (debug) {
63          LOGGER.log(Level.INFO, "verifying hostname " + hostname
64              + " with IgnoreHostName - will return true");
65        }
66        return true;
67      }
68  
69    }
70  
71    /**
72     * modifies the trustmanager
73     * 
74     * @author wf FIXME - move else where
75     */
76    private static class DefaultTrustManager implements X509TrustManager {
77  
78      @Override
79      public void checkClientTrusted(X509Certificate[] certificates,
80          String authType) throws CertificateException {
81      }
82  
83      @Override
84      public void checkServerTrusted(X509Certificate[] certificates,
85          String authType) throws CertificateException {
86      }
87  
88      @Override
89      public X509Certificate[] getAcceptedIssuers() {
90        return null;
91        // return new X509Certificate[0];
92      }
93    }

112   /**
113    * initialize this wiki
114    */
115   public void init() throws Exception {
116     // configure the SSLContext with a TrustManager
117     SSLContext ctx = SSLContext.getInstance("TLS");
118     if (ignoreCertificates) {
119       ctx.init(new KeyManager[0],
120           new TrustManager[] { new DefaultTrustManager() }, new SecureRandom());
121       SSLContext.setDefault(ctx);
122     }
123     HostnameVerifier hv = new IgnoreHostName();
124     HttpsURLConnection.setDefaultHostnameVerifier(hv);
125   }
126 
127   @Override
128   public void init(String siteurl, String scriptpath) throws Exception {
129     // make httpclient shut up see http://stackoverflow.com/a/15798443/1497139
130     org.apache.log4j.Logger.getLogger("org.apache.commons.httpclient")
131         .setLevel(org.apache.log4j.Level.ERROR);
132     org.apache.log4j.Logger.getLogger("httpclient.wire.header")
133         .setLevel(org.apache.log4j.Level.WARN);
134     org.apache.log4j.Logger.getLogger("httpclient.wire.content")
135         .setLevel(org.apache.log4j.Level.WARN);
136     java.util.logging.Logger.getLogger("org.apache.http.wire")
137         .setLevel(java.util.logging.Level.FINEST);
138     java.util.logging.Logger.getLogger("org.apache.http.headers")
139         .setLevel(java.util.logging.Level.FINEST);
140     super.init(siteurl, scriptpath);
141   }

181   /**
182    * log me in with the configured user
183    * 
184    * @throws Exception
185    */
186   public void login() throws Exception {
187     WikiUser wuser = WikiUser.getUser(getWikiid(), getSiteurl());
188     if (wuser == null) {
189       throw new Exception(
190           "user for " + getWikiid() + "(" + getSiteurl() + ") not configured");
191     }
192     // wiki.setDebug(true);
193     try {
194       Login login = login(wuser.getUsername(), wuser.getPassword());
195       LOGGER.log(Level.INFO, this.siteurl + this.scriptPath + this.apiPath + ":"
196           + login.getResult());
197       if (!"Success".equals(login.getResult())) {
198         throw new Exception("login for '" + wuser.getUsername() + "' at '"
199             + getWikiid() + "(" + this.getSiteurl() + this.getScriptPath()
200             + ")' failed: " + login.getResult());
201       }
202     } catch (javax.net.ssl.SSLHandshakeException she) {
203       String msg = "login via SSL  to " + this.getSiteurl() + " failed\n";
204       msg += "Exception: " + she.getMessage();
205       throw new Exception(msg);
206     }
207   }
